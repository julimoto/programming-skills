import ephem

def get_moon_phase_percentage(date): 
    moon = ephem.Moon() 
    moon.compute(date) 
    return moon.phase 

# Apply the moon phase function to the 'Start Date' 
df['Moon Phase Percentage'] = df['Start Date'].apply(get_moon_phase_percentage) 

# Function to get moon phase code 
def get_moon_phase_code(percentage): 
    if percentage < 1.84566 or percentage >= 98.15434: return 'NM' # New Moon 
    elif percentage < 22.16: return 'WC' # Waxing Crescent
    elif percentage < 27.16: return 'FQ' # First Quarter 
    elif percentage < 50.84: return 'WG' # Waxing Gibbous 
    elif percentage < 72.83: return 'FM' # Full Moon 
    elif percentage < 77.83: return 'WG' # Waning Gibbous 
    elif percentage < 99.99: return 'LQ' # Last Quarter 
    else: return 'WC' # Waning Crescent 
    
# Apply the moon phase code function to the 'Moon Phase Percentage' 
df['Moon Phase Code'] = df['Moon Phase Percentage'].apply(get_moon_phase_code)
